\section{Model of resilience}

\begin{frame}{Estimating resilience}
\beamerdefaultoverlayspecification{}
\small
Following \citet{sun_estimating_2021}, I specify
\begin{equation*}
Y_{i,t}^{SD} = \alpha_i + \lambda_t + \sum_{e \neq \infty}\sum_{k \neq -1} \delta_{e,k} Z_{i,t}^{e,k} + \varepsilon_{i,t}.
\end{equation*}
\begin{itemize}
  \item $Y_{i,t}^{SD}$ is pull requests merged, standardized across the five most recent pre-periods.
  \item $Z_{i,t}^{e,k}$ is the standard cohort $e$ treatment indicator for event time $k$.
  \begin{itemize}
    \item Treatment: Experiencing 1 abrupt important member departure
  \end{itemize}
  \item $\delta_{e,k}$ identifies average resilience under parallel trends and no anticipation.
  \begin{itemize}
    \item $e$: departure cohort; $k$: periods since departed member's last appearance
  \end{itemize}
  \item $\alpha_i$ and $\lambda_t$ are organization and calendar-time fixed effects.
\end{itemize}
\end{frame}

\begin{frame}{Plausibility of identifying assumptions}
\small
\begin{itemize}
  \item<1-> \focus<1>{Parallel trends}
  \begin{itemize}
    \item<1-> \focus<1>{Abrupt departures are unrelated to underlying organizational trends.}
    \item<1-> \focus<1>{Survey evidence from the literature shows that abrupt departures are driven by events unrelated to OSS such as job changes, graduation, or family changes.}
    \item<1-> \focus<1>{These reflect transitions in life, not responses to latent organizational decline or other organization unobservables.}
  \end{itemize}
  \item<2-> \focus<2>{No anticipation}
  \begin{itemize}
    \item<2-> \focus<2>{Violations of no anticipation mean organizations start adjusting months prior to departure dates.}
    \item<2-> \focus<2>{With six-month periods, prior-period adjustments seem unlikely; for example, tech-industry notice periods are typically around two weeks.}
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Flaws of single-practice median splits?}
\small
\begin{itemize}
  \item<1-> \focus<1>{They ignore interactions across practices.}
  \item<1-> \focus<1>{Evaluating interaction effects via subsetting quickly becomes infeasible and creates multiple-testing concerns.}
  \item<2-> \focus<2>{Solution: Machine learning!}
  \begin{itemize}
    \item<2-> \focus<2>{Capable of making predictions using models that incorporate interactions among features}
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Modeling heterogeneity in resilience}
\small
% TODO: Add an appendix slide describing how this is estimated, following the
% steps in source/lib/event_study_forest.R.
I reformulate the event study as a partially linear model with two key changes:
\begin{equation*}
Y_{i,t}^{SD} = \alpha_i + f_t(X_i) + \sum_{e \neq \infty}\sum_{k \neq -1} \delta_{e,k}(X_i) Z_{i,t}^{e,k} + \varepsilon_{i,t}
\end{equation*}
\begin{itemize}
  \item<2-> \focus<2>{Treatment effects vary with practices through $\delta_{e,k}(X_i)$.}
  \item<2-> \focus<2>{A flexible baseline term, $f_t(X_i)$, replaces calendar-time fixed effects.}
  \item<3-> \focus<3>{Under conditional parallel trends and no anticipation, $\delta_{e,k}(X_i)$ identifies conditional average resilience for organizations with pre-period practices $X_i$.}
  \item<3-> \focus<3>{Using $\delta_{e,k}(X_i)$, I will predict each organization's resilience out-of-sample from their pre-period practice scores.}
\end{itemize}
\end{frame}


\begin{frame}{Predicting out-of-sample resilience using practices}
\small
\begin{enumerate}
  \item<1-> \focus<1>{I partition the sample into 10 equally-sized folds.}
  \begin{itemize}
    \item<1-> \focus<1>{I estimate $\hat{\delta}_{e,k}(X_i)$ on 9 folds using a generalized random forest adapted to a panel setting.}
    \item<1-> \focus<1>{I predict $\hat{\delta}_{e,k}(X_{-i})$ for organizations in the held-out fold using their coarsened practice scores.}
  \end{itemize}
  \item<1-> \focus<1>{I repeat this to obtain predictions of resilience for all organizations, using just their practices}
  \item<1-> \focus<1>{I classify organizations whose predicted resilience exceeds the median as high resilience; all others are low resilience.}
\end{enumerate}
\end{frame}
