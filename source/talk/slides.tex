\section{Model of resilience}

\begin{frame}{Estimating resilience}
\small
Following \citet{sun_estimating_2021}, I specify
\begin{equation*}<1->
Y_{i,t}^{SD} = \alpha_i + \lambda_t + \sum_{e \neq \infty}\sum_{k \neq -1} \delta_{e,k} Z_{i,t}^{e,k} + \varepsilon_{i,t}.
\end{equation*}
\begin{itemize}
  \item<2-> \focus<2>{$Y_{i,t}^{SD}$ is standardized across the five most recent pre-periods, so coefficients are in standard deviation units.}
  \item<3-> \focus<3>{$Z_{i,t}^{e,k}$ is an indicator for whether organization $i$ was treated at time $e$ and is observed at event time $k$.}
  \item<4-> \focus<4>{$\alpha_i$ and $\lambda_t$ are organization and calendar-time fixed effects.}
  \item<5-> \focus<5>{$\delta_{e,k}$ identifies treatment cohort $e$-specific resilience $k$ periods after departure under parallel trends and no anticipation.}
\end{itemize}
\end{frame}

\begin{frame}{Plausibility of identifying assumptions}
\small
\begin{itemize}
  \item<1-> \focus<1>{Parallel trends}
  \begin{itemize}
    \item<1-> \focus<1>{Survey evidence from \citet{miller_why_2019} suggests that abrupt departures are driven by events unrelated to organizational unobservables, such as job changes, graduation, marriage, and children.}
    \item<1-> \focus<1>{These reflect transitions in life, not obvious responses to latent organizational decline or dissatisfaction with project leadership.}
  \end{itemize}
  \item<2-> \focus<2>{No anticipation}
  \begin{itemize}
    \item<2-> \focus<2>{A violation would require organizations to change behavior months before the departure date.}
    \item<2-> \focus<2>{With six-month periods, that seems unlikely; for example, tech-industry notice periods are typically around two weeks.}
  \end{itemize}
\end{itemize}
\end{frame}

\section{Predicting resilience using practices}

\begin{frame}[label=main-single]{Single practices have limited predictive power}
\small
\begin{itemize}
  \item I classify organizations as high or low adopters of each practice category based on whether their principal-component score is above or below the median.
  \item I then estimate resilience separately for the high- and low-adoption groups.
  \item No single practice category systematically predicts increased resilience.
\end{itemize}
\end{frame}

\begin{frame}{Flaws of single-practice median splits?}
\small
\begin{itemize}
  \item<1-> \focus<1>{They ignore nonlinearities and interactions across practices.}
  \item<2-> \focus<2>{Evaluating interaction effects via subsetting quickly becomes infeasible and creates multiple-testing concerns.}
  \item<3-> \focus<3>{Machine learning allows me to model nonlinearities and interactions without enumerating them individually.}
\end{itemize}
\end{frame}

\begin{frame}{Modeling heterogeneity in resilience}
\small
% TODO: Add an appendix slide describing how this is estimated, following the
% steps in source/lib/event_study_forest.R.
I reformulate the event study as a partially linear model with two key changes:
\begin{equation*}<1->
Y_{i,t}^{SD} = \alpha_i + f_t(X_i) + \sum_{e \neq \infty}\sum_{k \neq -1} \delta_{e,k}(X_i) Z_{i,t}^{e,k} + \varepsilon_{i,t}
\end{equation*}
\begin{itemize}
  \item<2-> \focus<2>{Treatment effects vary with observables through $\delta_{e,k}(X_i)$.}
  \item<3-> \focus<3>{A flexible baseline term, $f_t(X_i)$, replaces calendar-time fixed effects.}
  \item<4-> \focus<4>{Under conditional parallel trends and no anticipation, $\delta_{e,k}(X_i)$ identifies conditional average resilience for organizations in cohort $e$ at event time $k$ with pre-period practices $X_i$.}
  \item<5-> \focus<5>{I estimate this using a generalized random forest adapted to panel-data settings.}
\end{itemize}
\end{frame}


\begin{frame}{Predicting resilience using practices}
\small
\begin{enumerate}
  \item<1-> \focus<1>{I split the sample into 10 folds.}
  \item<2-> \focus<2>{I use nine folds to estimate $\delta_{e,k}(X_i)$ and then predict resilience in the held-out fold.}
  \begin{itemize}
    \item<2-> \focus<2>{Predictions for the held-out fold use only pre-period organizational practices $X_i$.}
  \end{itemize}
  \item<3-> \focus<3>{I repeat this across folds and classify organizations above and below the median pooled doubly robust prediction as high and low resilience.}
  \item<4-> \focus<4>{I then estimate separate event studies for the predicted high- and low-resilience groups.}
\end{enumerate}
\end{frame}
